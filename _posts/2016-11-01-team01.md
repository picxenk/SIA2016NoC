## Code

// 가장 의미가 있는 코드의 부분을 적습니다.
function Ball() {
  this.location = createVector(0, 0);
  this.velocity = createVector(0, 0);

  this.move = function() {
    location.add(velocity);
  }

  this.show = function() {
    ellipse(location.x, location.y, 10, 10);
  }
}
